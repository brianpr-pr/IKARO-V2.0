drop database if exists e_commerce_copy;
CREATE DATABASE e_commerce_copy
CHARACTER SET utf8
COLLATE utf8_unicode_ci;

use e_commerce_copy;
create table productos (
	producto_id varchar(12) PRIMARY KEY,
    nombre varchar(20),
    descripcion varchar(35),
    precio double(8,2),
    categoria varchar(25),
    fecha_agregacion date,
    nombre_archivo varchar(40)
);

create table productos (
	producto_id varchar(12) PRIMARY KEY,
    nombre varchar(20),
    descripcion varchar(35),
    precio double(8,2),
    categoria varchar(25),
    fecha_agregacion date,
    nombre_archivo varchar(40)
);


create table carro (
	carro_id varchar(12) PRIMARY KEY,
    cantidad int,
    descripcion varchar(35),
    precio_total double(8,2),
	producto_id varchar(12) ,
    FOREIGN KEY (producto_id) REFERENCES productos(producto_id)
);

create table usuarios (
	usuario_id varchar(12) PRIMARY KEY,
    nombre varchar(20),
    apellidos varchar(30),
    direccion varchar(40),
	fecha_nacimiento date,
    carro_id varchar(12),
    FOREIGN KEY (carro_id) REFERENCES carro(carro_id)
);
